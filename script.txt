// ================================// Smooth navigation active link// ================================const sections = document.querySelectorAll("section");const navLinks = document.querySelectorAll("nav a");window.addEventListener("scroll", () => {    let current = "";    sections.forEach(section => {        const sectionTop = section.offsetTop - 120;        if (window.scrollY >= sectionTop) {            current = section.getAttribute("id");        }    });    navLinks.forEach(link => {        link.classList.remove("active");        if (link.getAttribute("href") === "#" + current) {            link.classList.add("active");        }    });});.activity{padding:28px 0;border-bottom:1px solid #e6e6e6;}.activity-header{display:flex;justify-content:space-between;align-items:baseline;margin-bottom:8px;gap:20px;}.activity-header h4{margin:0;font-size:1.25rem;color:#1f5fa8;}.activity-date{color:#777;font-size:.95rem;white-space:nowrap;}.location{font-style:italic;color:#666;margin-bottom:10px;}.activity p{line-height:1.7;}
